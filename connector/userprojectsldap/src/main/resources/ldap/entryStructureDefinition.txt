ou=assignments
    => cn=${assignmentName}
        => cn=project 
          => org-openengsb-string: ${projectName}
        => cn=user
          => org-openengsb-string: ${username}
        => ou=permissions
            => cn=${permissionName}
        => ou=roles
            => cn=${roleName}
        
ou=permissions
    => cn=${permissionName}
        => cn=component
          => org-openengsb-string: ${componentName}
        => cn=action
          => org-openengsb-string: ${actionName}

ou=projects
    => cn=${projectName}
        => ou=attributes
            => cn=${attributeName}
              => org-openengsb-string: ${attributeValues-combined-by-separator}
    
ou=roles
    => cn=${roleName}
        => ou=permissions
            => cn=${permissionName}
        => ou=roles
            => cn=${rolename}
            
ou=users
    => cn=${username}
        => ou=attributes
            => cn=${attributeName}
              => org-openengsb-string: ${attributeValues-combined-by-separator}
        => ou=credentials
            => cn=${credentialType} with value as attribute string value
              => org-openengsb-string: ${credentialValue}
